# 6. Rendering Strategies (SSG, SSR, ISR, CSR, Streaming)

## Overview & Architecture

Next.js offers five core rendering strategies, each optimized for different use cases. In the App Router (Next.js 13+), these strategies are determined by your data fetching patterns, route segment configuration, and use of dynamic functions rather than explicit method names like `getStaticProps` or `getServerSideProps`.

```
┌───────────────────────────────────────────────────────────────────────┐
│                    RENDERING STRATEGIES SPECTRUM                       │
│                                                                       │
│  ◄── More Static                                    More Dynamic ──►  │
│                                                                       │
│  ┌─────┐   ┌─────┐   ┌─────┐   ┌───────────┐   ┌─────┐   ┌─────┐  │
│  │ SSG │   │ ISR │   │ PPR │   │ Streaming │   │ SSR │   │ CSR │  │
│  │     │   │     │   │     │   │   SSR     │   │     │   │     │  │
│  └─────┘   └─────┘   └─────┘   └───────────┘   └─────┘   └─────┘  │
│                                                                       │
│  Build    Build +    Static    Server +         Server    Client      │
│  time     periodic   shell +   progressive      per      per         │
│  only     regen      dynamic   streaming        request  request     │
│                                                                       │
│  TTFB:    TTFB:     TTFB:     TTFB: Fast       TTFB:    TTFB:       │
│  Fastest  Fast      Fast      FCP: Progressive  Slower   Fast(empty) │
│  FCP:     FCP:      FCP:                        FCP:     FCP:        │
│  Fastest  Fast      Fast                        Slower   Slowest     │
└───────────────────────────────────────────────────────────────────────┘
```

### Next.js 15 Rendering Defaults

```
┌──────────────────────────────────────────────────────────────────┐
│                 WHAT MAKES A ROUTE DYNAMIC?                       │
│                                                                   │
│  A route becomes DYNAMIC if it uses ANY of these:                │
│                                                                   │
│  1. cookies()           — Reads request cookies                   │
│  2. headers()           — Reads request headers                   │
│  3. searchParams        — Reads URL query parameters              │
│  4. connection()        — Signals dynamic rendering intent        │
│  5. fetch(url, { cache: 'no-store' })  — Uncached fetch          │
│  6. fetch(url) with no options  — Default in Next.js 15!         │
│  7. export const dynamic = 'force-dynamic'                        │
│                                                                   │
│  A route is STATIC if:                                            │
│  • No dynamic functions are used                                  │
│  • All fetches use force-cache or revalidate                      │
│  • export const dynamic = 'force-static'                          │
│  • generateStaticParams() is used                                 │
└──────────────────────────────────────────────────────────────────┘
```

## Q1. What is Static Site Generation (SSG) and how do you implement it in the App Router? (Beginner)

**Scenario:** You're building a marketing website with 20 pages (home, about, pricing, etc.) that rarely change. You want the fastest possible load times and minimal server costs.

**Answer:**

In the App Router, a route is statically generated by default when it doesn't use any dynamic functions and all data fetching is cached.

```typescript
// app/about/page.tsx — Automatically static (no dynamic functions)
export default function AboutPage() {
  return (
    <div className="max-w-4xl mx-auto p-8">
      <h1 className="text-4xl font-bold">About Us</h1>
      <p className="mt-4 text-lg text-gray-600">
        We are a team of engineers building the future of web development.
      </p>
    </div>
  );
}
// This page is generated at build time as static HTML.
// No server needed at runtime — served directly from CDN.
```

**SSG with data fetching:**

```typescript
// app/pricing/page.tsx — Static with cached data
export default async function PricingPage() {
  // Using force-cache makes this fetch cached at build time
  const res = await fetch('https://api.example.com/pricing', {
    cache: 'force-cache',
  });
  const plans = await res.json();

  return (
    <div className="grid grid-cols-3 gap-8 max-w-6xl mx-auto p-8">
      {plans.map((plan: { id: string; name: string; price: number; features: string[] }) => (
        <div key={plan.id} className="border rounded-xl p-6 shadow-sm">
          <h2 className="text-xl font-bold">{plan.name}</h2>
          <p className="text-3xl font-bold mt-2">${plan.price}/mo</p>
          <ul className="mt-4 space-y-2">
            {plan.features.map((feature: string) => (
              <li key={feature} className="flex items-center gap-2">
                <span className="text-green-500">✓</span>
                {feature}
              </li>
            ))}
          </ul>
        </div>
      ))}
    </div>
  );
}
```

**SSG with dynamic routes using `generateStaticParams`:**

```typescript
// app/blog/[slug]/page.tsx
import { notFound } from 'next/navigation';

// Tell Next.js which slugs to pre-render at build time
export async function generateStaticParams() {
  const res = await fetch('https://api.example.com/posts', {
    cache: 'force-cache',
  });
  const posts = await res.json();

  return posts.map((post: { slug: string }) => ({
    slug: post.slug,
  }));
}

export default async function BlogPost({
  params,
}: {
  params: Promise<{ slug: string }>;
}) {
  const { slug } = await params;
  const res = await fetch(`https://api.example.com/posts/${slug}`, {
    cache: 'force-cache',
  });

  if (!res.ok) notFound();

  const post = await res.json();

  return (
    <article className="prose max-w-3xl mx-auto p-8">
      <h1>{post.title}</h1>
      <time className="text-gray-500">{post.publishedAt}</time>
      <div dangerouslySetInnerHTML={{ __html: post.contentHtml }} />
    </article>
  );
}

// Control behavior for slugs NOT in generateStaticParams
export const dynamicParams = true; // true = generate on-demand, false = 404
```

## Q2. What is Server-Side Rendering (SSR) and when does a route become dynamic in Next.js 15? (Beginner)

**Scenario:** You're building a user dashboard that shows personalized data based on the logged-in user's cookies. The page must always show fresh data.

**Answer:**

SSR renders the page on the server for every incoming request. In the App Router, a route becomes server-rendered when it uses dynamic functions or uncached data fetching.

```typescript
// app/dashboard/page.tsx — Dynamic (SSR) because it uses cookies()
import { cookies } from 'next/headers';
import { db } from '@/lib/db';

export default async function DashboardPage() {
  const cookieStore = await cookies();
  const sessionToken = cookieStore.get('session')?.value;

  if (!sessionToken) {
    redirect('/login');
  }

  const user = await db.user.findFirst({
    where: { sessions: { some: { token: sessionToken } } },
    include: {
      projects: { orderBy: { updatedAt: 'desc' }, take: 5 },
      notifications: { where: { read: false }, take: 10 },
    },
  });

  return (
    <div className="p-6">
      <h1 className="text-2xl font-bold">Welcome back, {user?.name}</h1>

      <section className="mt-8">
        <h2 className="text-lg font-semibold">Recent Projects</h2>
        <ul className="mt-4 space-y-2">
          {user?.projects.map((project) => (
            <li key={project.id} className="p-4 border rounded-lg hover:bg-gray-50">
              <a href={`/projects/${project.id}`}>{project.name}</a>
              <span className="text-sm text-gray-500 ml-2">
                Updated {project.updatedAt.toLocaleDateString()}
              </span>
            </li>
          ))}
        </ul>
      </section>

      <section className="mt-8">
        <h2 className="text-lg font-semibold">
          Notifications ({user?.notifications.length})
        </h2>
        {/* ... notification list */}
      </section>
    </div>
  );
}
```

**All the triggers that make a route dynamic:**

```typescript
// Trigger 1: cookies()
import { cookies } from 'next/headers';
const cookieStore = await cookies();  // → DYNAMIC

// Trigger 2: headers()
import { headers } from 'next/headers';
const headersList = await headers();  // → DYNAMIC

// Trigger 3: searchParams
export default async function Page({
  searchParams,
}: {
  searchParams: Promise<{ q: string }>;
}) {
  const { q } = await searchParams;  // → DYNAMIC
}

// Trigger 4: connection()
import { connection } from 'next/server';
await connection();  // → DYNAMIC (explicit opt-in)

// Trigger 5: Uncached fetch (Next.js 15 default!)
const data = await fetch('https://api.example.com/data');  // → DYNAMIC

// Trigger 6: Route segment config
export const dynamic = 'force-dynamic';  // → DYNAMIC
```

```

│                    SSR REQUEST FLOW                            │
│                                                               │
│  Browser ──► CDN/Edge ──► Origin Server                       │
│                               │                               │
│                    ┌──────────▼──────────┐                    │
│                    │  1. Parse request    │                    │
│                    │  2. Read cookies     │                    │
│                    │  3. Query database   │                    │
│                    │  4. Render React     │                    │
│                    │  5. Generate HTML    │                    │

│                                                               │

```

## Q3. What is Incremental Static Regeneration (ISR) and how does time-based revalidation work? (Beginner)

**Scenario:** Your blog has 5,000 posts. Building all pages at build time takes 30 minutes. You want pages to be static for performance but update within minutes when content changes.

**Answer:**

ISR gives you the best of both worlds: static performance with periodic freshness. Pages are generated at build time (or on first request) and then regenerated in the background when the revalidation period expires.

```typescript
// app/blog/[slug]/page.tsx — ISR with 60-second revalidation

// Revalidate this page every 60 seconds
export const revalidate = 60;

// Pre-build the most popular 200 posts
export async function generateStaticParams() {
  const posts = await fetch('https://cms.example.com/api/posts?limit=200', {
    cache: 'force-cache',
  }).then((r) => r.json());

  return posts.map((post: { slug: string }) => ({ slug: post.slug }));
}

// Allow on-demand generation for posts not pre-built
export const dynamicParams = true;

export default async function BlogPostPage({
  params,
}: {
  params: Promise<{ slug: string }>;
}) {
  const { slug } = await params;

  const post = await fetch(`https://cms.example.com/api/posts/${slug}`, {
    next: { revalidate: 60 }, // Also configurable per-fetch
  }).then((r) => r.json());

  if (!post) return notFound();

  return (
    <article className="prose max-w-3xl mx-auto p-8">
      <h1>{post.title}</h1>
      <div className="flex items-center gap-4 text-gray-500 mt-2">
        <span>{post.author.name}</span>
        <time>{new Date(post.publishedAt).toLocaleDateString()}</time>
      </div>
      <div dangerouslySetInnerHTML={{ __html: post.contentHtml }} />
    </article>
  );
}
```

```
┌──────────────────────────────────────────────────────────────────┐
│                ISR STALE-WHILE-REVALIDATE FLOW                    │
│                                                                   │
│  Timeline:                                                        │
│                                                                   │
│  0s ──────── 60s ──────── 61s ──────── 62s ─────── 120s          │
│  │           │            │            │            │             │
│  ▼           ▼            ▼            ▼            ▼             │
│  BUILD       Visitor A    Visitor B    Visitor C    Visitor D     │
│  │           │            │            │            │             │
│  Generate    Serve        Serve STALE  Serve NEW   Serve cached  │
│  page v1     cached v1    + trigger    page v2     page v2       │
│              (instant)    background   (instant)   (instant)     │
│                           regeneration                            │
│                           │                                       │
│                           └─► Fetches fresh data                  │
│                               Renders new HTML                    │
│                               Stores as v2                        │
│                                                                   │
│  KEY: Visitor B gets stale data but triggers the refresh.         │
│       Visitor C (and all subsequent) get the fresh version.       │
│       NO visitor ever waits for data fetching!                    │
└──────────────────────────────────────────────────────────────────┘
```

**ISR with on-demand revalidation (hybrid approach):**

```typescript
// Combine time-based and on-demand revalidation
// Time-based: Safety net — data refreshes every 5 minutes at most
export const revalidate = 300;

// On-demand: Immediate refresh when CMS publishes
// app/api/revalidate/route.ts
import { revalidatePath } from 'next/cache';
import { NextRequest, NextResponse } from 'next/server';

export async function POST(request: NextRequest) {
  const { slug, secret } = await request.json();

  if (secret !== process.env.REVALIDATION_SECRET) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  revalidatePath(`/blog/${slug}`);

  return NextResponse.json({ revalidated: true });
}
```

## Q4. What is Client-Side Rendering (CSR) and when is it appropriate in a Next.js app? (Beginner)

**Scenario:** You're building an interactive stock trading dashboard with real-time price updates via WebSocket. The data changes every millisecond and has no SEO value.

**Answer:**

CSR renders content entirely in the browser using JavaScript. The server sends a minimal HTML shell, and React takes over to fetch data and render the UI.

```typescript
// app/trading/page.tsx — Server shell with client-rendered content
import { TradingDashboard } from '@/components/TradingDashboard';

export const metadata = {
  title: 'Trading Dashboard',
};

export default function TradingPage() {
  return (
    <div className="h-screen bg-gray-950">
      {/* This component renders entirely on the client */}
      <TradingDashboard />
    </div>
  );
}
```

```typescript
// components/TradingDashboard.tsx
'use client';

import { useEffect, useState, useRef } from 'react';

interface StockPrice {
  symbol: string;
  price: number;
  change: number;
  changePercent: number;
  volume: number;
}

export function TradingDashboard() {
  const [prices, setPrices] = useState<Map<string, StockPrice>>(new Map());
  const [connected, setConnected] = useState(false);
  const wsRef = useRef<WebSocket | null>(null);

  useEffect(() => {
    const ws = new WebSocket(process.env.NEXT_PUBLIC_WS_URL!);

    ws.onopen = () => {
      setConnected(true);
      ws.send(JSON.stringify({
        action: 'subscribe',
        symbols: ['AAPL', 'GOOGL', 'MSFT', 'AMZN', 'TSLA'],
      }));
    };

    ws.onmessage = (event) => {
      const update: StockPrice = JSON.parse(event.data);
      setPrices((prev) => new Map(prev).set(update.symbol, update));
    };

    ws.onclose = () => setConnected(false);

    wsRef.current = ws;

    return () => ws.close();
  }, []);

  return (
    <div className="p-4 text-white">
      <div className="flex items-center gap-2 mb-4">
        <div className={`h-2 w-2 rounded-full ${connected ? 'bg-green-500' : 'bg-red-500'}`} />
        <span className="text-sm">{connected ? 'Live' : 'Disconnected'}</span>
      </div>

      <div className="grid grid-cols-5 gap-4">
        {Array.from(prices.values()).map((stock) => (
          <div key={stock.symbol} className="bg-gray-900 rounded-lg p-4">
            <h3 className="font-bold text-lg">{stock.symbol}</h3>
            <p className="text-2xl font-mono">${stock.price.toFixed(2)}</p>
            <p className={`text-sm ${stock.change >= 0 ? 'text-green-400' : 'text-red-400'}`}>
              {stock.change >= 0 ? '+' : ''}{stock.change.toFixed(2)} ({stock.changePercent.toFixed(2)}%)
            </p>
          </div>
        ))}
      </div>
    </div>
  );
}
```

**When to use CSR:**

| Use CSR when... | Don't use CSR when... |
|---|---|
| Real-time data (WebSockets) | SEO matters |
| No SEO requirement | First load speed is critical |
| Highly interactive (charts, maps) | Content is mostly text |
| User-specific data only | Accessibility is primary concern |
| Behind authentication | Initial render must be meaningful |

```
┌──────────────────────────────────────────────────────────────┐
│                    CSR LOADING TIMELINE                        │
│                                                               │
│  0ms   ─── Request page ──────────────────────────            │
│  50ms  ─── Receive HTML shell (empty body) ───────            │
│  200ms ─── Download & parse JavaScript ───────────            │
│  500ms ─── React hydrates, renders loading state ─  ← FCP    │
│  800ms ─── Fetch data from API ───────────────────            │
│  1200ms ── Render data into DOM ──────────────────  ← LCP    │
│  1500ms ── Interactive ───────────────────────────  ← TTI    │
│                                                               │
│  Problem: 500ms+ of blank/loading screen                      │
│  Solution: Use Streaming SSR + Suspense instead               │
│            (see Q7: Streaming & Suspense Patterns)            │
└──────────────────────────────────────────────────────────────┘
```

## Q5. How do you decide which rendering strategy to use for different types of pages? (Beginner)

**Scenario:** You're building a full e-commerce platform with these pages: product listing, product detail, user dashboard, checkout, blog, admin panel. Which strategy should each use?

**Answer:**

```
┌──────────────────────────────────────────────────────────────────────┐
│              RENDERING STRATEGY DECISION MATRIX                       │
├──────────────────────┬────────────┬─────────────────────────────────┤
│  Page                │ Strategy   │ Reasoning                        │
├──────────────────────┼────────────┼─────────────────────────────────┤
│  Home page           │ ISR (60s)  │ SEO critical, changes weekly     │
│  Product listing     │ ISR (300s) │ SEO critical, prices change daily│
│  Product detail      │ ISR (60s)  │ SEO critical, stock changes      │
│  Search results      │ SSR        │ Depends on query (searchParams)  │
│  User dashboard      │ Streaming  │ Personalized, multiple data      │
│  Checkout            │ SSR        │ Cart state, payment security     │
│  Blog listing        │ ISR (3600s)│ SEO critical, rarely changes     │
│  Blog post           │ SSG + ISR  │ SEO critical, mostly static      │
│  Admin panel         │ CSR        │ No SEO, highly interactive       │
│  Login/Register      │ SSG        │ No data, just forms              │
│  About/Contact       │ SSG        │ No data, purely static           │
│  User profile (pub)  │ ISR (300s) │ SEO useful, user content changes │
│  Settings            │ SSR        │ Personalized, requires auth      │
│  API documentation   │ SSG        │ Static content, MDX              │
│  Real-time chat      │ CSR        │ WebSocket, no SEO                │
│  Order history       │ Streaming  │ Personalized, paginated data     │
│  Notifications       │ CSR        │ Real-time updates, no SEO        │
└──────────────────────┴────────────┴─────────────────────────────────┘
```

**Decision flowchart:**

```
START: What kind of page is this?
  │
  ├── Does SEO matter?
  │   │
  │   ├── NO → Is it real-time/interactive?
  │   │         │
  │   │         ├── YES → CSR (Client-Side Rendering)
  │   │         └── NO  → SSR with streaming
  │   │
  │   └── YES → Does the content change?
  │             │
  │             ├── NEVER → SSG (Static Site Generation)
  │             │
  │             ├── RARELY (hours/days)
  │             │   └── ISR with long revalidation (1-24 hours)
  │             │
  │             ├── FREQUENTLY (minutes)
  │             │   └── ISR with short revalidation (30-300 seconds)
  │             │
  │             ├── PER-USER (personalized)
  │             │   └── SSR with streaming + Suspense
  │             │
  │             └── REAL-TIME (seconds)
  │                 └── SSR + client-side updates (WebSocket/polling)
  │
  └── Is data needed from multiple sources?
      │
      ├── YES → Streaming SSR with parallel Suspense boundaries
      └── NO  → Simple SSR or ISR
```

**Implementation example — mixing strategies on one page:**

```typescript
// app/products/[slug]/page.tsx
// STRATEGY: ISR for product info + Streaming for reviews + CSR for live inventory
import { Suspense } from 'react';
import { LiveInventory } from '@/components/LiveInventory';

// ISR: Revalidate product info every 5 minutes
export const revalidate = 300;

export async function generateStaticParams() {
  // Pre-generate top 500 products at build time
  const products = await fetch('https://api.store.com/top-products?limit=500', {
    cache: 'force-cache',
  }).then((r) => r.json());

  return products.map((p: { slug: string }) => ({ slug: p.slug }));
}

export default async function ProductPage({
  params,
}: {
  params: Promise<{ slug: string }>;
}) {
  const { slug } = await params;

  // Cached fetch — part of the ISR page
  const product = await fetch(`https://api.store.com/products/${slug}`, {
    next: { revalidate: 300, tags: [`product-${slug}`] },
  }).then((r) => r.json());

  return (
    <div>
      {/* STATIC (ISR): Product info — cached, fast */}
      <ProductInfo product={product} />

      {/* CSR: Live inventory — real-time, client-side */}
      <LiveInventory productId={product.id} initialStock={product.stock} />

      {/* STREAMING: Reviews — server-rendered, streamed in */}
      <Suspense fallback={<ReviewsSkeleton />}>
        <ProductReviews productId={product.id} />
      </Suspense>
    </div>
  );
}

// This component makes the reviews section stream separately
async function ProductReviews({ productId }: { productId: string }) {
  const reviews = await fetch(
    `https://api.store.com/products/${productId}/reviews`,
    { cache: 'no-store' } // Always fresh reviews
  ).then((r) => r.json());

  return (
    <section className="mt-8">
      <h2 className="text-xl font-bold">Reviews ({reviews.length})</h2>
      {reviews.map((review: { id: string; author: string; rating: number; text: string }) => (
        <div key={review.id} className="border-b py-4">
          <div className="flex items-center gap-2">
            <span className="font-medium">{review.author}</span>
            <span>{'★'.repeat(review.rating)}{'☆'.repeat(5 - review.rating)}</span>
          </div>
          <p className="mt-1 text-gray-600">{review.text}</p>
        </div>
      ))}
    </section>
  );
}
```

## Q6. How do `dynamic`, `revalidate`, and `fetchCache` route segment configs interact to determine rendering behavior? (Intermediate)

**Scenario:** You've set `export const revalidate = 60` on a page but it's still rendering dynamically on every request. You need to understand how route segment configs work together.

**Answer:**

```typescript
// Route Segment Config Options
export const dynamic = 'auto' | 'force-dynamic' | 'error' | 'force-static';
export const revalidate = false | 0 | number; // seconds
export const fetchCache = 'auto' | 'default-cache' | 'only-cache' |
  'force-cache' | 'default-no-store' | 'only-no-store' | 'force-no-store';
export const runtime = 'nodejs' | 'edge';
export const preferredRegion = 'auto' | 'global' | 'home' | string[];
```

**How they interact:**

```
┌──────────────────────────────────────────────────────────────────┐
│           ROUTE SEGMENT CONFIG PRIORITY                           │
│                                                                   │
│  1. dynamic = 'force-dynamic'                                     │
│     └── OVERRIDES EVERYTHING → Always SSR, ignores revalidate     │
│                                                                   │
│  2. dynamic = 'force-static'                                      │
│     └── Forces static → cookies/headers return empty              │
│         Dynamic functions won't make route dynamic                │
│                                                                   │
│  3. dynamic = 'error'                                             │
│     └── Forces static → ERRORS if dynamic functions used          │
│         Good for catching accidental dynamic usage                │
│                                                                   │
│  4. dynamic = 'auto' (default)                                    │
│     └── Rendering determined by usage:                            │
│         ├── Uses dynamic functions? → SSR                         │
│         ├── Uses no-store fetch? → SSR                            │
│         ├── Uses revalidate fetch? → ISR                          │
│         └── All cached? → SSG                                     │
│                                                                   │
│  REVALIDATE:                                                      │
│  ├── false → Cache indefinitely (SSG)                             │
│  ├── 0 → Always re-render (equivalent to force-dynamic)          │
│  └── number → ISR with that interval                             │
│                                                                   │
│  The LOWEST revalidate value wins across all fetches in a route:  │
│  fetch(url1, { next: { revalidate: 3600 } })  → 1 hour           │
│  fetch(url2, { next: { revalidate: 60 } })    → 1 minute         │
│  Route revalidation: 60 seconds (lowest wins)                     │
└──────────────────────────────────────────────────────────────────┘
```

**Common pitfalls and fixes:**

```typescript
// PITFALL 1: revalidate set but cookies() used → DYNAMIC

  const cookieStore = await cookies(); // ← This overrides revalidate!
  // Page is SSR, not ISR

  const data = await fetch('https://api.com/data', {
    next: { revalidate: 3600 },
  });

  const user = await fetch('https://api.com/user'); // ← no-store default in Next.js 15!
  // This makes the entire route dynamic!

  // FIX: Add cache option to ALL fetches
  const user2 = await fetch('https://api.com/user', {
    next: { revalidate: 3600 },
  });
}

// PITFALL 3: Layout revalidate vs page revalidate
// app/blog/layout.tsx

// app/blog/[slug]/page.tsx

// but the layout data revalidates every 1 hour.
// The lowest revalidate in the RENDERING path determines
// when the ROUTE is re-rendered.
```

## Q10. How do you implement a mixed rendering strategy for an e-commerce category page with filters? (Intermediate)

**Scenario:** Your category page `/products/shoes` shows products with filters (size, color, price range). The initial page should be static/ISR for SEO, but applying filters should dynamically update results without a full page reload.

```typescript

  params: Promise<{ category: string }>;
  searchParams: Promise<{ [key: string]: string | string[] | undefined }>;

  const { category } = await params;
  const filters = await searchParams;

  const categoryData = await fetch(

  const filterOptions = await fetch(
    `https://api.store.com/categories/${category}/filters`,

          <FilterSidebar
            options={filterOptions}
            currentFilters={filters}
            category={category}
          />
        </aside>

        {/* DYNAMIC when filters active, STATIC when no filters */}
        <main className="flex-1">
          <Suspense
            key={JSON.stringify(filters)} // Re-trigger Suspense when filters change
            fallback={<ProductGridSkeleton />}
          >
            <FilteredProductGrid
              category={category}
              filters={filters}
              hasActiveFilters={hasActiveFilters}
            />
          </Suspense>
        </main>

// This component handles both static and dynamic rendering
async function FilteredProductGrid({
  category,
  filters,
  hasActiveFilters,
}: {
  category: string;
  filters: Record<string, string | string[] | undefined>;
  hasActiveFilters: boolean;
}) {
  const queryString = new URLSearchParams(
    Object.entries(filters).reduce<Record<string, string>>((acc, [key, val]) => {
      if (val) acc[key] = Array.isArray(val) ? val.join(',') : val;
      return acc;
    }, {})
  ).toString();

  const url = `https://api.store.com/products?category=${category}${
    queryString ? `&${queryString}` : ''
  }`;

  // If filters are active → fresh data (dynamic)
  // If no filters → cached data (static/ISR)
  const products = await fetch(url, {
    ...(hasActiveFilters
      ? { cache: 'no-store' as const }
      : { next: { revalidate: 300, tags: [`category-${category}`] } }),
  }).then((r) => r.json());
```

```typescript
// components/ErrorBoundary.tsx
'use client';

import React, { Component, type ReactNode } from 'react';

interface Props {
  children: ReactNode;
  fallback: ReactNode;
  onError?: (error: Error) => void;
}

interface State {
  hasError: boolean;
  error: Error | null;
}

export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false, error: null };
  }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error) {
    this.props.onError?.(error);
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback;
    }
    return this.props.children;
  }
}
```

```typescript
  const hostname = request.headers.get('host') || '';
  const subdomain = hostname.split('.')[0];
```

## Q16. How do you implement dynamic rendering at the edge for geo-personalized content? (Advanced)

**Scenario:** Your e-commerce site needs to show different currencies, languages, and product availability based on the visitor's location. You want edge-level rendering for speed but the content varies per region.

**Answer:**

```typescript
// middleware.ts — Geo-detection and routing
import { NextRequest, NextResponse } from 'next/server';

const SUPPORTED_REGIONS: Record<string, { currency: string; locale: string }> = {
  US: { currency: 'USD', locale: 'en-US' },
  GB: { currency: 'GBP', locale: 'en-GB' },
  EU: { currency: 'EUR', locale: 'en-EU' },
  JP: { currency: 'JPY', locale: 'ja-JP' },
  IN: { currency: 'INR', locale: 'en-IN' },
};

export function middleware(request: NextRequest) {
  const country = request.geo?.country || 'US';

  // Map country to region
  let region = 'US';
  if (['GB'].includes(country)) region = 'GB';
  else if (['DE', 'FR', 'IT', 'ES', 'NL'].includes(country)) region = 'EU';
  else if (['JP'].includes(country)) region = 'JP';
  else if (['IN'].includes(country)) region = 'IN';

  const regionConfig = SUPPORTED_REGIONS[region];

  const response = NextResponse.next();

  // Set geo headers for downstream components
  response.headers.set('x-user-country', country);
  response.headers.set('x-user-region', region);
  response.headers.set('x-user-currency', regionConfig.currency);
  response.headers.set('x-user-locale', regionConfig.locale);

  return response;
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';

export function middleware(request: NextRequest) {
  // Only for A/B test routes
  if (!request.nextUrl.pathname.startsWith('/pricing')) {
    return NextResponse.next();
  }

  const existingVariant = request.cookies.get('ab-pricing')?.value;

  if (existingVariant) {
    // User already assigned — set header and continue
    const response = NextResponse.next();
    response.headers.set('x-ab-variant', existingVariant);
    return response;

  // Assign variant (50/50 split)
  const variant = Math.random() < 0.5 ? 'control' : 'variant-b';

  const response = NextResponse.next();
  response.cookies.set('ab-pricing', variant, {
    maxAge: 60 * 60 * 24 * 30, // 30 days
    httpOnly: true,
    sameSite: 'lax',
  });
  response.headers.set('x-ab-variant', variant);

  return response;
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';

import crypto from 'crypto';

const SHOPIFY_WEBHOOK_SECRET = process.env.SHOPIFY_WEBHOOK_SECRET!;

// Verify Shopify webhook signature
function verifyShopifyWebhook(body: string, signature: string): boolean {
  const hmac = crypto
    .createHmac('sha256', SHOPIFY_WEBHOOK_SECRET)
    .update(body, 'utf-8')
    .digest('base64');

  return crypto.timingSafeEqual(
    Buffer.from(hmac),
    Buffer.from(signature)
  );
}

export async function POST(request: NextRequest) {
  const body = await request.text();
  const signature = request.headers.get('x-shopify-hmac-sha256') || '';
  const topic = request.headers.get('x-shopify-topic') || '';

  if (!verifyShopifyWebhook(body, signature)) {
    return NextResponse.json({ error: 'Invalid signature' }, { status: 401 });
  }

  const data = JSON.parse(body);

  console.log(`[Shopify Webhook] ${topic}`, { id: data.id });

  switch (topic) {
    case 'products/update':
    case 'products/create':
```

```typescript
// public/sw.js — Service Worker with network-first + cache-fallback strategy

const CACHE_NAME = 'app-v1';
const STATIC_CACHE = 'static-v1';

// Assets to pre-cache during install
const PRECACHE_ASSETS = [
  '/offline',
  '/_next/static/css/app.css',
];

// Install: pre-cache critical assets
self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(STATIC_CACHE).then((cache) => {
      return cache.addAll(PRECACHE_ASSETS);
    })
  );
  self.skipWaiting();
});

// Activate: clean up old caches
self.addEventListener('activate', (event) => {
  event.waitUntil(
    caches.keys().then((keys) => {
      return Promise.all(
        keys
          .filter((key) => key !== CACHE_NAME && key !== STATIC_CACHE)
          .map((key) => caches.delete(key))
      );
    })
  );
  self.clients.claim();
});

// Fetch: Network-first with cache fallback
self.addEventListener('fetch', (event) => {
  const { request } = event;
  const url = new URL(request.url);

  // Skip non-GET requests
  if (request.method !== 'GET') return;

  // Skip API routes and Server Actions
  if (url.pathname.startsWith('/api/') || request.headers.get('next-action')) {
    return;
  }

  // Static assets: cache-first
  if (url.pathname.startsWith('/_next/static/')) {
    event.respondWith(
      caches.match(request).then((cached) => {
        return cached || fetch(request).then((response) => {
          const clone = response.clone();
          caches.open(STATIC_CACHE).then((cache) => cache.put(request, clone));
          return response;
        });
      })
    );
    return;
  }

  // HTML pages: network-first with cache fallback
  if (request.headers.get('accept')?.includes('text/html')) {
    event.respondWith(
      fetchWithTimeout(request, 5000)
        .then((response) => {
          // Cache successful responses
          if (response.ok) {
            const clone = response.clone();
            caches.open(CACHE_NAME).then((cache) => cache.put(request, clone));
          }
          return response;
        })
        .catch(() => {
          // Network failed — try cache
          return caches.match(request).then((cached) => {
            if (cached) {
              // Return cached version with offline indicator header
              return new Response(cached.body, {
                headers: {
                  ...Object.fromEntries(cached.headers),
                  'x-served-from': 'cache',
                },
              });
            }
            // No cache — show offline page
            return caches.match('/offline');
          });
        })
    );
    return;
  }
});

// Fetch with timeout for poor network conditions
function fetchWithTimeout(request, timeoutMs) {
  return new Promise((resolve, reject) => {
    const timer = setTimeout(() => {
      reject(new Error('Fetch timeout'));
    }, timeoutMs);

    fetch(request).then(
      (response) => {
        clearTimeout(timer);
        resolve(response);
      },
      (error) => {
        clearTimeout(timer);
        reject(error);
      }
    );
  });
}
```

```typescript
'use client';

import { useEffect, useState } from 'react';

type ConnectionStatus = 'online' | 'slow' | 'offline' | 'cached';

export function NetworkStatusIndicator() {
  const [status, setStatus] = useState<ConnectionStatus>('online');

  useEffect(() => {
    function updateStatus() {
      if (!navigator.onLine) {
        setStatus('offline');
        return;
      }

      const connection = (navigator as Navigator & {
        connection?: { effectiveType: string; downlink: number };
      }).connection;

      if (connection) {
        if (connection.effectiveType === '2g' || connection.downlink < 0.5) {
          setStatus('slow');
        } else {
          setStatus('online');
        }
      }
    }

    // Check if page was served from cache
    const entries = performance.getEntriesByType('navigation') as PerformanceNavigationTiming[];
    if (entries[0]?.transferSize === 0) {
      setStatus('cached');
    }

    window.addEventListener('online', updateStatus);
    window.addEventListener('offline', updateStatus);
    updateStatus();

    return () => {
      window.removeEventListener('online', updateStatus);
      window.removeEventListener('offline', updateStatus);
    };
  }, []);

  if (status === 'online') return null;

  const statusConfig = {
    slow: { bg: 'bg-yellow-500', text: 'Slow connection — showing cached content where available' },
    offline: { bg: 'bg-red-500', text: 'You are offline — showing cached content' },
    cached: { bg: 'bg-blue-500', text: 'Showing cached version — content may not be current' },
  };

  const config = statusConfig[status];

  return (
    <div className={`${config.bg} text-white text-sm text-center py-2 px-4`}>
      {config.text}
    </div>
  );
}
```
